### Buy Fighters ###
country_event = {
	id = freeponies.2
	immediate = {log = "[Root.GetName]: event freeponies.2"}#Auto-logging
	title = freeponies.2.t
	desc = freeponies.2.d
	picture = GFX_report_event_fighters

	is_triggered_only = yes


	option = { #Yes
		name = freeponies.2.a
		ai_chance = {
			factor = 10
		}
		FROM = { country_event = freeponies.3 }
		effect_tooltip = {
			FROM = {
				if = {
					limit = {
						not = {
							has_tech = early_fighter
						}
					}
					set_technology = { early_fighter = 1 }
				}
				add_equipment_to_stockpile = { type = fighter_equipment_0 amount = 25 producer = ROOT }
			}
		}
	}

	option = { #No
		name = freeponies.2.b
		ai_chance = {
			factor = 0
		}
		FROM = { country_event = freeponies.4 }

	}
}
